      if (DEBUG) {
        Log.e("Swipe has been called but swiping is not enabled");
      }
      return;
    }
    if (viewHolder.itemView.getParent()!= mRecyclerView) {
      Log.e("Swipe has been called with a view holder which is not a child of "
        + "the RecyclerView controlled by this ItemTouchHelper.");
      return;
    }
    if (mActionState!= ACTION_STATE_SWIPE) {
      return;
    }
    if (mSelected!= null && mSelected == viewHolder) {
      return;
    }
    if (mSelected!= null) {
      endRecoverAnimation(mSelected, false); // this may push it into pending cleanup list.
      if (mPendingCleanup.remove(mSelected.itemView)) {
        mCallback.clearView(mRecyclerView, mSelected);
      }
    }
    select(viewHolder, ACTION_STATE_SWIPE);
    mCallback.onSwiped(viewHolder, direction);
  }
