/**
 * The Class ConcurrentMapCache.
 *
 * @param <K> the key type
 * @param <V> the value type
 */
public class ConcurrentMapCache<K, V> implements Cache<K, V> {

    /** The cache. */
    private Map<K, V> cache;

    /**
     * Instantiates a new concurrent map cache.
     */
    public ConcurrentMapCache() {
        cache = new ConcurrentHashMap<K, V>();
    }

    /**
     * Instantiates a new concurrent map cache.
     *
     * @param initialCapacity the initial capacity
     */
    public ConcurrentMapCache(int initialCapacity) {
        cache = new ConcurrentHashMap<K, V>(initialCapacity);
    }

    /**
     * Instantiates a new concurrent map cache.
     *
     * @param initialCapacity the initial capacity
     * @param loadFactor the load factor
     */
    public ConcurrentMapCache(int initialCapacity, float loadFactor) {
        cache = new ConcurrentHashMap<K, V>(initialCapacity, loadFactor);
    }

    /**
     * Instantiates a new concurrent map cache.
     *
     * @param initialCapacity the initial capacity
     * @param loadFactor the load factor
     * @param concurrencyLevel the concurrency level
     */
    public ConcurrentMapCache(int initialCapacity, float loadFactor, int concurrencyLevel) {
        cache = new ConcurrentHashMap<K, V>(initialCapacity, loadFactor, concurrencyLevel);
    }

    public V get(K key) {
        return cache.get(key);
